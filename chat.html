Chatbot b√°sico que utiliza listas para almacenar posibles entradas del usuario y respuestas predefinidas. Usa el operador 'in' para verificar si la entrada coincide con alguna opci√≥n y random.choice() para seleccionar respuestas variadas. 

# Ejemplo chatbot con listas basico
import random

# Listas de saludos y respuestas
saludos = ["hola", "que tal", "buenos dias", "buenas tardes", "buenas noches"]
respuesta_saludos = ["hola, como estas?",
                     "¬øcomo puedo ayudarte?",
                     "hola, como te va?",
                     "que hay?"]

despedidas = ["adios", "bye", "gracias", "nos vemos", "hasta luego", "chao"]

respuesta_despedida = ["Adios hasta pronto",
                       "Fue un placer ayudarte",
                       "Cuidate",
                       "Nos vemos luegooo",
                       "Byeee"]

p = ["cuantas champions tiene el madrid"]
pr = ["el mejor jugador"]
pre = ["jared o mancilla"]

r = ["robadas 15", "10"]
re = ["messi", "henry martin"]
res = ["tecate"]

# Mensaje de bienvenida del chatbot
print("Hola soy Chatoo, tu amigo personal de confianza, si deseas "
      "terminar escribe 'salir'.")

while True:
    usuario = input("yo:").lower().strip()

    if usuario in saludos:
        print("Amigo Personal:", random.choice(respuesta_saludos))
    elif usuario in p:
        print("Amigo Personal:", random.choice(r))
    elif usuario in pr:
        print("Amigo Personal:", random.choice(re))
    elif usuario in pre:
        print("Amigo Personal:", random.choice(res))
    elif usuario in despedidas:
        print("Amigo Personal:", random.choice(respuesta_despedida))
        break
